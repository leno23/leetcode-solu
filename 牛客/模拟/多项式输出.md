let arr = []
require('readline').createInterface({
    input: process.stdin,
    output: process.stdout
}).on('line', line => {
    arr.push(line)
    if (arr.length < 2) return
    let n = +arr[0], cs = arr[1].split(' ').map(Number)
    let res = ''
    for (let i = n; i >= 0; i--) {
        let a = cs[n - i];
        if (a == 0) continue
        if (i != n && a > 0) res += '+';
        if (Math.abs(a) > 1 || i == 0) res += a;
        if (a == -1 && i != 0) res += '-';
        if (i > 1) res += "x^" + i;
        if (i == 1) res += 'x';
    }
    console.log(res)

})
