/**
 * @param {number} n
 * @return {number}
 */
var magicalString = function (n) {
    let s = '122', i = 2
    while (s.length < n) {
        // 根据s最后一个字母确定下一个字符
        let next = s[s.length - 1] ^ 3
        s += next
        if (s[i] == '2') s += next
        i++
    }
    let res = 0
    for (let i = 0; i < n; i++) if (s[i] == '1') res++
    return res
};
