/**
 * @param {number[]} nums
 * @return {number}
 */
var subArrayRanges = function (nums) {
    let res = 0,
        n = nums.length;
    for (let i = 0; i < n; i++) {
        let min = nums[i];
        let max = nums[i];
        for (let j = i + 1; j < n; j++) {
            max = Math.max(max, nums[j]);
            min = Math.min(min, nums[j]);
            res += max - min;
        }
    }
    return res;
};
