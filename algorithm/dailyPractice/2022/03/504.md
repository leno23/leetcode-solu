504. 七进制数

给定一个整数 num，将其转化为 7 进制，并以字符串形式输出。

示例 1:

输入: num = 100
输出: "202"

```js
/**
 * @param {number} num
 * @return {string}
 */
var convertToBase7 = function (num) {
    if (num == 0) return "0";
    let arr = [];
    let neg = false;
    if (num < 0) (neg = true), (num = -num);
    // 逐位取模法，进行进制转化
    while (num) {
        arr.push(num % 7);
        num = (num / 7) | 0;
    }
    neg && arr.push("-");
    let res = "";
    for (let i = arr.length - 1; i >= 0; i--) res += arr[i];
    return res;
};
```
