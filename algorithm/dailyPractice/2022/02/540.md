/**
 * @param {number[]} nums
 * @return {number}
 */
var singleNonDuplicate = function (nums) {
    let n = nums.length;
    let l = 0,
        r = n - 1;
    while (l < r) {
        let mid = (l + r) >> 1;
        if (mid % 2 == 0) {
            if (mid + 1 < n && nums[mid] == nums[mid + 1]) l = mid + 1;
            else r = mid;
        } else {
            if (mid - 1 >= 0 && nums[mid - 1] == nums[mid]) l = mid + 1;
            else r = mid;
        }
    }
    return nums[r];
};
