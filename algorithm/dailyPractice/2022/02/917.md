/**
 * @param {string} s
 * @return {string}
 */
var reverseOnlyLetters = function (s) {
    let cs = [...s];
    let str = [];
    const isLow = (c) =>
        (c.charCodeAt() >= 97 && c.charCodeAt() <= 122) ||
        (c.charCodeAt() >= 65 && c.charCodeAt() <= 90);
    for (let i = 0; i < cs.length; i++) {
        if (isLow(cs[i])) {
            str.push(cs[i]);
            cs[i] = "";
        }
    }
    let len = str.length;
    for (let i = 0; i < cs.length; i++) {
        if (cs[i] == "") cs[i] = str[--len];
    }
    return cs.join("");
};
