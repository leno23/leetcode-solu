/**
 * @param {number[]} nums
 * @return {number}
 */
var countQuadruplets = function (nums) {
    let n = nums.length;
    let ans = 0;
    let cnt = new Map();
    // d-b=
    for (let b = n - 3; b >= 1; --b) {
        for (let d = b + 2; d < n; ++d) {
            cnt.set(
                nums[d] - nums[b + 1],
                (cnt.get(nums[d] - nums[b + 1]) || 0) + 1
            );
        }
        for (let a = 0; a < b; ++a) {
            ans += cnt.get(nums[a] + nums[b]) || 0;
        }
    }
    return ans;
};
