/**
 * @param {number} n
 * @param {number} k
 * @return {number}
 */
var findKthNumber = function (n, k) {
    const getCount = (prefix, n) => {
        let next = prefix + 1,
            res = 0;
        for (let i = prefix; i <= n; i *= 10, next *= 10) {
            res += Math.min(next, n + 1) - i;
        }
        return res;
    };
    let p = 1,
        prefix = 1;
    while (p < k) {
        let cnt = getCount(prefix, n);
        if (cnt + p > k) {
            prefix *= 10;
            p++;
        } else {
            prefix++;
            p += cnt;
        }
    }
    return prefix;
};
