/**
 * @param {number[]} nums
 * @return {number}
 */
var countDistinctIntegers = function (nums) {
    const reverse = num => {
        let res = 0
        while (num) {
            res = res * 10 + num % 10
            num = num / 10 | 0
        }
        return res
    }
    let s = new Set(nums)
    for (let x of nums) {
        s.add(reverse(x))
    }
    return s.size
};
